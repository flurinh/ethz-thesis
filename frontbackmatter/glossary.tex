\chapter*{Glossary}

\section*{Symbols}%
\vskip -2em
\begin{tabularx}{\textwidth}{lX}
  $\lambda_\text{max}$ & absorption maximum (wavelength of peak absorption) \\
  $\Delta\lambda$ & spectral shift ($\lambda_\text{max}^\text{AT} - \lambda_\text{max}^\text{11-cis}$) \\
  $E$ & photon energy ($E = hc/\lambda$) \\
  $R^2$ & coefficient of determination \\
  $h$ & Planck constant \\
  $c$ & speed of light in vacuum \\
\end{tabularx}
